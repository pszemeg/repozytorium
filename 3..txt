#include <stdio.h>
#define ROW 8
#define COL 3
main()
{

                        };
     int tab2[ROW][COL]={
        {2,3,4},

                        };
    for(a=0; a<ROW; a++)
            {

             }
     for(a=0; a<ROW; a++)

                printf("\n");

                }
            }
}
